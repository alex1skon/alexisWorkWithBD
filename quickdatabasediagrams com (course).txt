exercises
--
exercise_id int PK IDENTITY
exercise varchar(50) NULL
approaches int NULL

guest
--
guest_id int PK IDENTITY
fullName varchar(50) NULL
birth date NULL
hostel varchar(50) NULL
course int NULL
classGroup varchar(10) NULL
trainer_id int INDEX NULL FK >- trainer.trainer_id

musclesgroup
--
musclesGroup_id int PK IDENTITY
musclesGroup varchar(50) NULL
exercise_id int INDEX NULL FK >- exercises.exercise_id

timetable
--
timetable_id int PK IDENTITY
Datetime datetime NULL

trainer
--
trainer_id int PK IDENTITY

phoneNumber varchar(13) NULL

--
trainingProgram_id int PK IDENTITY

ended datetime NULL
